  <!-- Main Script -->
  <script>
    (function() {
      'use strict';

      // DOM Elements
      const grid = document.getElementById('docs-grid');
      const filterContainer = document.getElementById('filter-container');
      const countElement = document.getElementById('resource-count');

      // State
      let currentFilter = 'all';

      // Extract unique categories
      function getCategories(data) {
        const cats = new Set();
        data.forEach(item => {
          if (item.cat) cats.add(item.cat);
        });
        return ['all', ...Array.from(cats)];
      }

      // Render filter tags
      function renderFilters(categories) {
        filterContainer.innerHTML = '';
        
        categories.forEach(cat => {
          const btn = document.createElement('button');
          btn.className = 'filter-tag' + (cat === currentFilter ? ' active' : '');
          btn.textContent = cat === 'all' ? 'All' : cat;
          btn.setAttribute('aria-pressed', cat === currentFilter);
          btn.addEventListener('click', () => handleFilter(cat));
          filterContainer.appendChild(btn);
        });
      }

      // Handle filter change
      function handleFilter(category) {
        currentFilter = category;
        
        // Update active states
        document.querySelectorAll('.filter-tag').forEach(btn => {
          const isActive = (btn.textContent.toLowerCase() === category.toLowerCase()) || 
                          (category === 'all' && btn.textContent === 'All');
          btn.classList.toggle('active', isActive);
          btn.setAttribute('aria-pressed', isActive);
        });

        renderGrid();
      }

      // Extract domain from URL
      function extractDomain(url) {
        try {
          const domain = new URL(url).hostname.replace('www.', '');
          return domain;
        } catch {
          return 'external';
        }
      }

      // Render single card
      function createCard(item, index) {
        const card = document.createElement('a');
        card.href = item.link;
        card.target = '_blank';
        card.rel = 'noopener noreferrer';
        card.className = 'doc-card';
        card.style.animationDelay = `${index * 0.05}s`;
        card.setAttribute('aria-label', `Visit ${item.name}`);

        card.innerHTML = `
          <div class="card-header">
            <h3 class="card-title">${escapeHtml(item.name)}</h3>
            <svg class="card-arrow" viewBox="0 0 24 24" fill="none" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
              <line x1="7" y1="17" x2="17" y2="7"/>
              <polyline points="7,7 17,7 17,17"/>
            </svg>
          </div>
          <p class="card-desc">${escapeHtml(item.desc)}</p>
          <div class="card-footer">
            <span class="card-category">${escapeHtml(item.cat || 'General')}</span>
            <span class="card-domain">${extractDomain(item.link)}</span>
          </div>
        `;

        return card;
      }

      // Escape HTML
      function escapeHtml(str) {
        const div = document.createElement('div');
        div.textContent = str;
        return div.innerHTML;
      }

      // Render grid
      function renderGrid() {
        if (typeof docsData === 'undefined' || !Array.isArray(docsData)) {
          grid.innerHTML = `
            <div class="empty-state">
              <svg viewBox="0 0 24 24" fill="none" stroke-width="1.5">
                <circle cx="12" cy="12" r="10"/>
                <line x1="12" y1="8" x2="12" y2="12"/>
                <line x1="12" y1="16" x2="12.01" y2="16"/>
              </svg>
              <p>No data available</p>
            </div>
          `;
          return;
        }

        // Filter data
        const filtered = currentFilter === 'all' 
          ? docsData 
          : docsData.filter(item => (item.cat || '').toLowerCase() === currentFilter.toLowerCase());

        // Update count
        countElement.textContent = filtered.length;

        // Clear and render
        grid.innerHTML = '';
        
        if (filtered.length === 0) {
          grid.innerHTML = `
            <div class="empty-state" style="grid-column: 1 / -1;">
              <svg viewBox="0 0 24 24" fill="none" stroke-width="1.5">
                <circle cx="11" cy="11" r="8"/>
                <line x1="21" y1="21" x2="16.65" y2="16.65"/>
              </svg>
              <p>No resources found in this category</p>
            </div>
          `;
          return;
        }

        filtered.forEach((item, index) => {
          grid.appendChild(createCard(item, index));
        });
      }

      // Initialize
      function init() {
        if (typeof docsData !== 'undefined' && Array.isArray(docsData)) {
          const categories = getCategories(docsData);
          renderFilters(categories);
          renderGrid();
        } else {
          renderGrid();
        }
      }

      // Run on DOM ready
      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
      } else {
        init();
      }
    })();
  </script>
