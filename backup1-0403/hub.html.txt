<script>
// Récupération des données depuis les fichiers externes chargés
const data = window.BOOKMARKS_DATA || [];
const TAGS = window.BOOKMARKS_TAGS || {};

// DOM Elements
const container = document.getElementById('cards-container');
const filterBar = document.getElementById('filter-bar');
const searchInput = document.getElementById('search-input');
const searchPrefix = document.getElementById('search-prefix');
const themeToggle = document.getElementById('theme-checkbox');
const alphabetNav = document.getElementById('alphabet-nav');
const cursor = document.querySelector('.custom-cursor');

// State
let currentCategory = 'all';
let searchQuery = '';

// Custom cursor (only for fine pointer devices)
const hasFinePointer = window.matchMedia('(pointer: fine)').matches;

if (hasFinePointer && cursor) {
  let cursorX = 0;
  let cursorY = 0;
  let targetX = 0;
  let targetY = 0;
  
  document.addEventListener('mousemove', (e) => {
    targetX = e.clientX;
    targetY = e.clientY;
  });
  
  function animateCursor() {
    const dx = targetX - cursorX;
    const dy = targetY - cursorY;
    
    cursorX += dx * 0.15;
    cursorY += dy * 0.15;
    
    const size = parseInt(getComputedStyle(document.documentElement).getPropertyValue('--cursor-size')) || 22;
    cursor.style.transform = `translate(${cursorX - size/2}px, ${cursorY - size/2}px)`;
    
    requestAnimationFrame(animateCursor);
  }
  animateCursor();
  
  const interactiveElements = document.querySelectorAll('button, a, .card, input, .theme-switch, .slider, .search-box, .filter-btn, .alphabet-link, .logo-mark');
  
  interactiveElements.forEach(el => {
    el.addEventListener('mouseenter', () => cursor.classList.add('hover'));
    el.addEventListener('mouseleave', () => cursor.classList.remove('hover'));
  });
}

// Clock
function updateClock() {
  const now = new Date();
  const timeEl = document.getElementById('clock-time');
  const dateEl = document.getElementById('clock-date');
  
  if (timeEl) {
    timeEl.textContent = now.toLocaleTimeString('en-US', { 
      hour: '2-digit', 
      minute: '2-digit', 
      hour12: false 
    });
  }
  
  if (dateEl) {
    dateEl.textContent = now.toLocaleDateString('en-US', { 
      weekday: 'short', 
      month: 'short', 
      day: 'numeric' 
    });
  }
}
setInterval(updateClock, 1000);
updateClock();

// Theme
if (themeToggle) {
  const savedTheme = localStorage.getItem('theme');
  if (savedTheme === 'light') {
    document.documentElement.setAttribute('data-theme', 'light');
    themeToggle.checked = true;
  }
  
  themeToggle.addEventListener('change', (e) => {
    const theme = e.target.checked ? 'light' : 'dark';
    document.documentElement.setAttribute('data-theme', theme);
    localStorage.setItem('theme', theme);
  });
}

// Build Alphabet Navigation
function buildAlphabetNav() {
  const categories = [...new Set(data.map(i => i.cat))].sort();
  
  if (categories.length === 0) {
    alphabetNav.style.display = 'none';
    return;
  }
  
  alphabetNav.innerHTML = categories.map(cat => {
    const letter = cat.charAt(0).toUpperCase();
    return `<a class="alphabet-link" data-target="${cat}" data-category="${cat}" href="#cat-${cat}" aria-label="Jump to ${cat}">${letter}</a>`;
  }).join('');
  
  document.querySelectorAll('.alphabet-link').forEach(link => {
    link.addEventListener('click', (e) => {
      e.preventDefault();
      const targetId = `cat-${link.dataset.target}`;
      const target = document.getElementById(targetId);
      
      if (target) {
        const offset = 120;
        const top = target.getBoundingClientRect().top + window.pageYOffset - offset;
        window.scrollTo({ top, behavior: 'smooth' });
      }
    });
  });
}

// Scroll Spy
function handleScroll() {
  const sections = document.querySelectorAll('.category-section');
  const scrollPos = window.pageYOffset;
  let current = '';
  
  sections.forEach(s => {
    if (scrollPos >= s.offsetTop - 180) {
      current = s.id.replace('cat-', '');
    }
  });
  
  document.querySelectorAll('.alphabet-link').forEach(link => {
    const isActive = link.dataset.target === current;
    link.classList.toggle('active', isActive);
    link.setAttribute('aria-current', isActive ? 'true' : 'false');
  });
}

let scrollTimeout;
window.addEventListener('scroll', () => {
  if (scrollTimeout) cancelAnimationFrame(scrollTimeout);
  scrollTimeout = requestAnimationFrame(handleScroll);
}, { passive: true });

// Create Card
function createCard(item) {
  const card = document.createElement('article');
  card.className = 'card';
  card.setAttribute('role', 'button');
  card.setAttribute('tabindex', '0');
  card.setAttribute('aria-label', `Open ${item.name}`);
  
  const tags = TAGS[item.name] || [];
  
  card.innerHTML = `
    <span class="card-category">${item.cat}</span>
    <h3 class="card-title">${item.name}</h3>
    <p class="card-desc">${item.desc || ''}</p>
    ${tags.length > 0 ? `
      <div class="tags" aria-label="Tags">
        ${tags.map(t => `<span class="tag">${t}</span>`).join('')}
      </div>
    ` : ''}
  `;
  
  const openLink = () => {
    if (item.link && item.link !== '#') {
      window.open(item.link, '_blank', 'noopener,noreferrer');
    }
  };
  
  card.addEventListener('click', openLink);
  card.addEventListener('keydown', (e) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      openLink();
    }
  });
  
  return card;
}

// Create No Results
function createNoResults() {
  const el = document.createElement('div');
  el.className = 'no-results';
  el.innerHTML = `
    <svg class="no-results-icon" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5">
      <circle cx="11" cy="11" r="8"/>
      <path d="m21 21-4.35-4.35"/>
      <path d="M8 8l6 6M14 8l-6 6"/>
    </svg>
    <p>No results found for <span class="highlight">"${searchQuery}"</span></p>
    <small>Try adjusting your search or filter</small>
  `;
  return el;
}

// Render
function render() {
  container.innerHTML = '';
  
  const filtered = data.filter(item => {
    const matchCat = currentCategory === 'all' || item.cat === currentCategory;
    const searchText = (item.name + ' ' + item.cat + ' ' + (item.desc || '') + ' ' + (TAGS[item.name] || []).join(' ')).toLowerCase();
    const matchSearch = searchText.includes(searchQuery.toLowerCase());
    return matchCat && matchSearch;
  });
  
  document.getElementById('count-visible').textContent = filtered.length;
  document.getElementById('count-total').textContent = data.length;
  
  searchPrefix.textContent = currentCategory === 'all' ? 'All:' : `${currentCategory}:`;
  
  if (filtered.length === 0) {
    container.appendChild(createNoResults());
    alphabetNav.style.display = 'none';
    return;
  }
  
  const showAlphabetNav = currentCategory === 'all' && searchQuery === '';
  alphabetNav.style.display = showAlphabetNav ? 'flex' : 'none';
  
  if (showAlphabetNav) {
    const groups = filtered.reduce((acc, item) => {
      (acc[item.cat] = acc[item.cat] || []).push(item);
      return acc;
    }, {});
    
    const sortedCategories = Object.keys(groups).sort();
    
    sortedCategories.forEach(cat => {
      const section = document.createElement('div');
      section.className = 'category-section';
      section.id = `cat-${cat}`;
      
      const itemCount = groups[cat].length;
      
      section.innerHTML = `
        <div class="category-header">
          ${cat}
          <span class="category-count">${itemCount}</span>
        </div>
      `;
      
      container.appendChild(section);
      
      groups[cat].forEach(item => {
        container.appendChild(createCard(item));
      });
    });
  } else {
    filtered.forEach(item => {
      container.appendChild(createCard(item));
    });
  }
}

// Init Filters
function initFilters() {
  const cats = ['all', ...new Set(data.map(i => i.cat))].sort((a, b) => {
    if (a === 'all') return -1;
    if (b === 'all') return 1;
    return a.localeCompare(b);
  });
  
  filterBar.innerHTML = cats.map(c => 
    `<button class="filter-btn ${c === 'all' ? 'active' : ''}" data-cat="${c}" aria-pressed="${c === 'all'}">${c}</button>`
  ).join('');
  
  filterBar.addEventListener('click', (e) => {
    const btn = e.target.closest('.filter-btn');
    if (!btn) return;
    
    const cat = btn.dataset.cat;
    
    if (currentCategory === cat && cat !== 'all') {
      currentCategory = 'all';
    } else {
      currentCategory = cat;
    }
    
    document.querySelectorAll('.filter-btn').forEach(b => {
      const isActive = b.dataset.cat === currentCategory;
      b.classList.toggle('active', isActive);
      b.setAttribute('aria-pressed', isActive);
    });
    
    render();
  });
}

// Search Handler
let searchTimeout;
searchInput.addEventListener('input', (e) => {
  clearTimeout(searchTimeout);
  searchTimeout = setTimeout(() => {
    searchQuery = e.target.value.trim();
    render();
  }, 150);
});

// Keyboard Navigation
document.addEventListener('keydown', (e) => {
  if (e.key === '/' && document.activeElement !== searchInput) {
    e.preventDefault();
    searchInput.focus();
  }
  
  if (e.key === 'Escape' && document.activeElement === searchInput) {
    searchInput.value = '';
    searchQuery = '';
    searchInput.blur();
    render();
  }
});

// Initialize
function init() {
  buildAlphabetNav();
  initFilters();
  render();
}

if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', init);
} else {
  init();
}
</script>
