// Alphabet Nav
function buildAlphabetNav() {
  const categories = [...new Set(data.map(i => i.cat))].sort();
  if (categories.length === 0) {
    alphabetNav.style.display = 'none';
    return;
  }
  alphabetNav.innerHTML = categories.map(cat => 
    `<a class="alphabet-link" data-target="${cat}" href="#cat-${cat}" aria-label="Jump to ${cat}">${cat.charAt(0).toUpperCase()}</a>`
  ).join('');
  
  alphabetNav.addEventListener('click', (e) => {
    const link = e.target.closest('.alphabet-link');
    if (!link) return;
    e.preventDefault();
    const target = document.getElementById(`cat-${link.dataset.target}`);
    if (target) window.scrollTo({ top: target.offsetTop - 120, behavior: 'smooth' });
  });
}

// Helper
function extractDomain(url) {
  try {
    const hostname = new URL(url).hostname;
    return hostname.replace('www.', '');
  } catch (e) {
    return 'link';
  }
}

// Create Card
function createCard(item) {
  const card = document.createElement('article');
  card.className = 'card';
  card.setAttribute('role', 'button');
  card.setAttribute('tabindex', '0');
  
  const allTags = TAGS[item.name] || [];
  const domain = extractDomain(item.link);
  const isSafety = item.cat === "Safety & Prevention";
  
  const isSearchActive = searchQuery.length > 0;
  const limit = 3;
  const visibleTags = isSearchActive ? allTags : allTags.slice(0, limit);
  const hiddenTags = isSearchActive ? [] : allTags.slice(limit);
  
  let iconSvg;
  if (isSafety) {
      iconSvg = `<svg class="card-heart" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
          <path d="M20.84 4.61a5.5 5.5 0 0 0-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 0 0-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 0 0 0-7.78z"></path>
      </svg>`;
  } else {
      iconSvg = `<svg class="card-arrow" viewBox="0 0 24 24" fill="none" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
          <line x1="7" y1="17" x2="17" y2="7"/>
          <polyline points="7,7 17,7 17,17"/>
      </svg>`;
  }
  
  let tagsHtml = visibleTags.map(t => `<span class="tag">${t}</span>`).join('');
  let showMoreHtml = '';
  if (hiddenTags.length > 0) {
    const blurredTagsHtml = hiddenTags.map(t => `<span class="tag tag-blur">${t}</span>`).join('');
    tagsHtml += blurredTagsHtml;
    showMoreHtml = `
      <div class="tag-show-more-wrapper">
        <button class="tag-show-more" onclick="event.stopPropagation(); this.closest('.tags-wrapper').classList.add('revealed')">+${hiddenTags.length} more</button>
      </div>`;
  }

  const tagsWrapperHtml = (allTags.length > 0) ? `
    <div class="tags-wrapper ${isSearchActive ? 'revealed' : ''}">
      <div class="tags">${tagsHtml}</div>
      ${showMoreHtml}
    </div>` : '';

  card.innerHTML = `
    <div class="card-header">
      <h3 class="card-title">${item.name}</h3>
      ${iconSvg}
    </div>
    <p class="card-desc">${item.desc || ''}</p>
    ${tagsWrapperHtml}
    <div class="card-footer">
      <span class="card-category">${item.cat}</span>
      <span class="card-domain">${domain}</span>
    </div>
  `;
  
  const openLink = () => {
    if (item.link && item.link !== '#') window.open(item.link, '_blank', 'noopener,noreferrer');
  };
  
  card.addEventListener('click', (e) => {
    if (e.target.closest('.tag-show-more')) return;
    openLink();
  });
  
  card.addEventListener('keydown', (e) => { if (e.key === 'Enter') openLink(); });
  return card;
}

function render() {
  container.innerHTML = '';

    const matchCat = currentCategory === 'all' || item.cat === currentCategory;
    const searchText = (item.name + ' ' + item.cat + ' ' + (item.desc || '') + ' ' + (TAGS[item.name] || []).join(' ')).toLowerCase();
    const matchSearch = searchText.includes(searchQuery.toLowerCase());
    return matchCat && matchSearch;
  });
  
  document.getElementById('count-visible').textContent = filtered.length;
  document.getElementById('count-total').textContent = data.length;
  searchPrefix.textContent = currentCategory === 'all' ? 'All:' : `${currentCategory}:`;
  
  const showFeatured = currentCategory === 'all' && searchQuery === '';
  
  // FEATURED SECTION
  if (showFeatured) {
    const featuredData = [
      { name: 'Docs', desc: 'Documentations', link: 'docs.html' },
      { name: 'Files', desc: 'Games, Softwares, Plugins', link: 'files.html' },
      { name: 'Jobs', desc: 'Career & Opportunities', link: 'jobs.html' },
      { name: 'Mention', desc: 'Highlights & Recognition', link: 'mentions.html' },
      { name: 'GitHub', desc: 'Repositories & Projects', link: 'https://github.com' },
      { name: 'Support', desc: 'Help & Assistance', link: 'support.html' }
    ];
    
    container.appendChild(createFeaturedSeparator());
    featuredData.forEach(item => container.appendChild(createSportCard(item)));
  }
  
  if (filtered.length === 0 && !showFeatured) {
    container.appendChild(createNoResults());
    alphabetNav.style.display = 'none';
    return;
  }
  
  // Logic for showing 6 cards then blur
  const shouldLimitCards = (currentCategory === 'all' && searchQuery === '');
  const visibleLimit = 6;
  const initialCards = shouldLimitCards ? filtered.slice(0, visibleLimit) : filtered;
  const remainingCards = shouldLimitCards ? filtered.slice(visibleLimit) : [];
  
  const showAlphabetNav = currentCategory === 'all' && searchQuery === '';
  alphabetNav.style.display = showAlphabetNav ? 'flex' : 'none';
  
  // RENDER INITIAL CARDS
  if (showAlphabetNav) {
    const groups = initialCards.reduce((acc, item) => {
      (acc[item.cat] = acc[item.cat] || []).push(item);
      return acc;
    }, {});
    
    Object.keys(groups).sort().forEach(cat => {
      const section = document.createElement('div');
      section.className = 'category-section';
      section.id = `cat-${cat}`;
      section.innerHTML = `<div class="category-header">${cat}<span class="category-count">${groups[cat].length}</span></div>`;
      container.appendChild(section);
      groups[cat].forEach(item => container.appendChild(createCard(item)));
    });
  } else {
    initialCards.forEach(item => container.appendChild(createCard(item)));
  }
  
  // RENDER REMAINING CARDS (BLURRED)
  if (remainingCards.length > 0 && !cardsExpanded) {
    // 1. Create the main wrapper (Grid Item)
    const blurWrapper = document.createElement('div');
    blurWrapper.className = 'cards-blur-wrapper';
    
    // 2. Create the blur content container
    const blurContent = document.createElement('div');
    blurContent.className = 'cards-blur-section';
    blurContent.id = 'cards-blur-container';
    
    const innerGrid = document.createElement('div');
    innerGrid.className = 'collapsible-container';
    
    if (showAlphabetNav) {
      const remainingGroups = remainingCards.reduce((acc, item) => {
        (acc[item.cat] = acc[item.cat] || []).push(item);
        return acc;
      }, {});
      
      Object.keys(remainingGroups).sort().forEach(cat => {
        const section = document.createElement('div');
        section.className = 'category-section';
        section.innerHTML = `<div class="category-header">${cat}<span class="category-count">${remainingGroups[cat].length}</span></div>`;
        innerGrid.appendChild(section);
        remainingGroups[cat].forEach(item => innerGrid.appendChild(createCard(item)));
      });
    } else {
      remainingCards.forEach(item => innerGrid.appendChild(createCard(item)));
    }
    
    blurContent.appendChild(innerGrid);
    
    // 3. Create button wrapper (Sibling of blur content)
    const btnWrapper = document.createElement('div');
    btnWrapper.className = 'show-more-cards-btn-wrapper';
    
    const btn = document.createElement('button');
    btn.className = 'show-more-cards-btn';
    btn.textContent = `Show ${remainingCards.length} More`;
    btn.onclick = () => {
      cardsExpanded = true;
      const blurSec = document.getElementById('cards-blur-container');
      if(blurSec) {
        blurSec.classList.add('expanded');
        setTimeout(() => {
           blurSec.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }, 100);
      }
      btnWrapper.style.display = 'none'; // Hide button after click
    };
    
    btnWrapper.appendChild(btn);
    
    // Assemble: Wrapper -> [BlurContent, Button]
    blurWrapper.appendChild(blurContent);
    blurWrapper.appendChild(btnWrapper);
    
    container.appendChild(blurWrapper);
    
  } else if (cardsExpanded && remainingCards.length > 0) {
     if (showAlphabetNav) {
        const groups = remainingCards.reduce((acc, item) => {
          (acc[item.cat] = acc[item.cat] || []).push(item);
          return acc;
        }, {});
        
        Object.keys(groups).sort().forEach(cat => {
          const section = document.createElement('div');
          section.className = 'category-section';
          section.innerHTML = `<div class="category-header">${cat}<span class="category-count">${groups[cat].length}</span></div>`;
          container.appendChild(section);
          groups[cat].forEach(item => container.appendChild(createCard(item)));
        });
     } else {
        remainingCards.forEach(item => container.appendChild(createCard(item)));
     }
  }
  
}

// Init Filters
function initFilters() {
  const cats = ['all', ...new Set(data.map(i => i.cat))].sort((a, b) => {
    if (a === 'all') return -1;
    if (b === 'all') return 1;
    return a.localeCompare(b);
  });
  
  filterBar.innerHTML = cats.map(c => `<button class="filter-btn ${c === 'all' ? 'active' : ''}" data-cat="${c}">${c}</button>`).join('');
  
  filterBar.addEventListener('click', (e) => {
    const btn = e.target.closest('.filter-btn');
    if (!btn) return;
    
    const cat = btn.dataset.cat;
    currentCategory = (currentCategory === cat && cat !== 'all') ? 'all' : cat;
    
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.toggle('active', b.dataset.cat === currentCategory));
    
    showLoader();
    setTimeout(() => {
      render();
      hideLoader();
    }, 200);
  });
}

// Search
let searchTimeout;
searchInput.addEventListener('input', (e) => {
  clearTimeout(searchTimeout);
  showLoader();
  searchTimeout = setTimeout(() => {
    searchQuery = e.target.value.trim();
    render();
    hideLoader();
  }, 300);
});

// Keyboard
document.addEventListener('keydown', (e) => {
  if (e.key === '/' && document.activeElement !== searchInput) {
    e.preventDefault();
    searchInput.focus();
  }
  if (e.key === 'Escape' && document.activeElement === searchInput) {
    searchInput.value = '';
    searchQuery = '';
    searchInput.blur();
    render();
  }
});

// Init
function init() {
  showLoader();
  setTimeout(() => {
    buildAlphabetNav();
    initFilters();
    render();
    hideLoader();
  }, 300);
}

if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', init);
} else {
  init();
}
