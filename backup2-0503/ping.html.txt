    <p class="footer-text">Outil d'inspection HTTP - Mesure de disponibilite et latence</p>
  </footer>
</div>

<script>
(function() {
  // DOM Elements
  const urlInput = document.getElementById('url-input');
  const pingBtn = document.getElementById('ping-btn');
  const emptyState = document.getElementById('empty-state');
  const loadingContainer = document.getElementById('loading-container');
  const resultCard = document.getElementById('result-card');
  const statusIndicator = document.getElementById('status-indicator');
  const statusCode = document.getElementById('status-code');
  const statusText = document.getElementById('status-text');
  const responseTime = document.getElementById('response-time');
  const resultUrl = document.getElementById('result-url');
  const statSize = document.getElementById('stat-size');
  const statType = document.getElementById('stat-type');
  const statServer = document.getElementById('stat-server');
  const statIps = document.getElementById('stat-ips');
  const headersToggle = document.getElementById('headers-toggle');
  const headersContent = document.getElementById('headers-content');
  const quickLinks = document.querySelectorAll('.quick-link');

  // State
  let isLoading = false;

  // HTTP Status Messages
  const statusMessages = {
    200: 'OK',
    201: 'Created',
    204: 'No Content',
    301: 'Moved Permanently',
    302: 'Found',
    304: 'Not Modified',
    400: 'Bad Request',
    401: 'Unauthorized',
    403: 'Forbidden',
    404: 'Not Found',
    405: 'Method Not Allowed',
    429: 'Too Many Requests',
    500: 'Internal Server Error',
    502: 'Bad Gateway',
    503: 'Service Unavailable',
    504: 'Gateway Timeout'
  };

  // Format bytes
  function formatBytes(bytes) {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
  }

  // Parse URL
  function parseUrl(input) {
    let url = input.trim();
    if (!url.startsWith('http://') && !url.startsWith('https://')) {
      url = 'https://' + url;
    }
    try {
      return new URL(url);
    } catch (e) {
      return null;
    }
  }

  // Show loading
  function showLoading() {
    isLoading = true;
    pingBtn.disabled = true;
    emptyState.style.display = 'none';
    resultCard.classList.remove('visible');
    loadingContainer.classList.add('active');
  }

  // Hide loading
  function hideLoading() {
    isLoading = false;
    pingBtn.disabled = false;
    loadingContainer.classList.remove('active');
  }

  // Show result
  function showResult(data) {
    resultCard.classList.add('visible');
    
    // Status
    const isSuccess = data.status >= 200 && data.status < 400;
    resultCard.classList.toggle('success', isSuccess);
    resultCard.classList.toggle('error', !isSuccess);
    
    statusIndicator.className = 'status-indicator ' + (isSuccess ? 'online' : 'offline');
    statusCode.textContent = data.status;
    statusCode.className = 'status-code ' + (isSuccess ? 'success' : 'error');
    statusText.textContent = statusMessages[data.status] || 'Unknown';
    
    // Time
    responseTime.textContent = data.time.toFixed(0);
    
    // URL
    resultUrl.textContent = data.url;
    
    // Stats
    statSize.textContent = formatBytes(data.size);
    statType.textContent = data.contentType || '---';
    statServer.textContent = data.server || '---';
    statIps.textContent = 'GET';
    
    // Headers
    headersContent.innerHTML = Object.entries(data.headers)
      .map(function(entry) {
        return '<div class="header-line"><span class="header-key">' + entry[0] + ':</span><span class="header-value">' + entry[1] + '</span></div>';
      })
      .join('');
  }

  // Show error
  function showError(error) {
    resultCard.classList.add('visible', 'error');
    statusIndicator.className = 'status-indicator offline';
    statusCode.textContent = 'ERR';
    statusCode.className = 'status-code error';
    statusText.textContent = error;
    responseTime.textContent = '--';
    resultUrl.textContent = urlInput.value || '---';
    statSize.textContent = '--';
    statType.textContent = '--';
    statServer.textContent = '--';
    statIps.textContent = '--';
    headersContent.innerHTML = '<div class="header-line"><span class="header-value">Aucun en-tete disponible</span></div>';
  }

  // Make request
  async function makeRequest() {
    const urlObj = parseUrl(urlInput.value);
    
    if (!urlObj) {
      showError('URL invalide');
      return;
    }

    showLoading();
    const startTime = performance.now();

    try {
      const controller = new AbortController();
      const timeoutId = setTimeout(function() { controller.abort(); }, 15000);

      const response = await fetch(urlObj.href, {
        method: 'GET',
        mode: 'cors',
        signal: controller.signal
      });

      clearTimeout(timeoutId);
      const endTime = performance.now();
      
      // Get response body size
      let size = 0;
      let contentType = '';
      
      try {
        const clone = response.clone();
        const blob = await clone.blob();
        size = blob.size;
      } catch (e) {
        size = 0;
      }

      // Headers
      const headers = {};
      response.headers.forEach(function(value, key) {
        headers[key] = value;
      });

      contentType = headers['content-type'] ? headers['content-type'].split(';')[0] : '---';

      showResult({
        status: response.status,
        time: endTime - startTime,
        url: urlObj.href,
        size: size,
        contentType: contentType,
        server: headers['server'] || '---',
        headers: headers
      });

    } catch (error) {
      if (error.name === 'AbortError') {
        showError('Timeout (15s)');
      } else if (error.message.includes('CORS') || error.message.includes('NetworkError')) {
        showError('CORS / Network Error');
      } else {
        showError('Connexion impossible');
      }
    }

    hideLoading();
  }

  // Event Listeners
  pingBtn.addEventListener('click', makeRequest);

  urlInput.addEventListener('keydown', function(e) {
    if (e.key === 'Enter' && !isLoading) {
      makeRequest();
    }
  });

  // Quick links
  quickLinks.forEach(function(link) {
    link.addEventListener('click', function(e) {
      e.preventDefault();
      urlInput.value = this.dataset.url;
      if (!isLoading) {
        makeRequest();
      }
    });
  });

  // Headers toggle
  headersToggle.addEventListener('click', function() {
    const isExpanded = headersContent.classList.contains('visible');
    headersContent.classList.toggle('visible');
    headersToggle.classList.toggle('expanded');
    headersToggle.setAttribute('aria-expanded', !isExpanded);
  });

  headersToggle.addEventListener('keydown', function(e) {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      headersToggle.click();
    }
  });

  // Focus input on load
  setTimeout(function() {
    urlInput.focus();
  }, 300);
})();
</script>

</body>
