  try { return new URL(url).hostname.replace('www.', ''); }

function createCard(item) {
  const card = document.createElement('article');
  card.className = 'card';
  card.setAttribute('tabindex', '0');
  const domain = extractDomain(item.link);
  
  card.innerHTML = `
    <div class="card-header">
      <h3 class="card-title">${item.name}</h3>
      <svg class="card-arrow" viewBox="0 0 24 24" fill="none" stroke-width="2"><line x1="7" y1="17" x2="17" y2="7"/><polyline points="7,7 17,7 17,17"/></svg>
    </div>
    <p class="card-desc">${item.desc || ''}</p>
    <div class="card-footer">
      <span class="card-category">${item.cat}</span>
      <span class="card-domain">${domain}</span>
    </div>
  `;
  
  const openLink = () => { if (item.link && item.link !== '#') window.open(item.link, '_blank', 'noopener,noreferrer'); };
  card.addEventListener('click', openLink);
  card.addEventListener('keydown', (e) => { if (e.key === 'Enter') openLink(); });

}

function render() {
  container.innerHTML = '';
  
  const filtered = data.filter(item => {
    const matchCat = activeCategories.size === 0 || activeCategories.has(item.cat);
    const searchContent = (item.name + ' ' + item.cat + ' ' + (item.desc || '')).toLowerCase();
    const matchSearch = searchContent.includes(searchQuery.toLowerCase());
    return matchCat && matchSearch;
  });
  
  document.getElementById('count-visible').textContent = filtered.length;
  document.getElementById('count-total').textContent = data.length;
  
  if (activeCategories.size === 0) searchPrefix.textContent = 'All:';
  else if (activeCategories.size === 1) searchPrefix.textContent = [...activeCategories][0] + ':';
  else searchPrefix.textContent = activeCategories.size + ' Cats:';
  
  const showFeatured = activeCategories.size === 0 && searchQuery === '';
  
  if (showFeatured) {
    const featuredData = [

      { name: 'Jobs', desc: 'Career', link: 'jobs.html' },
      { name: 'News', desc: 'Updates', link: 'news.html' },
      { name: 'GitHub', desc: 'Projects', link: 'https://github.com/bookmarks-cih' },

    ];
    
    const separator = document.createElement('div');

    container.appendChild(separator);
    
    featuredData.forEach(item => container.appendChild(createSportCard(item)));
  }
  
  if (filtered.length === 0 && !showFeatured) {
    const el = document.createElement('div');
    el.className = 'no-results';
    el.innerHTML = `<p>No results found for <span class="highlight">"${searchQuery}"</span></p>`;
    container.appendChild(el);
    alphabetNav.style.display = 'none';
    return;
  }
  
  const shouldLimitCards = (activeCategories.size === 0 && searchQuery === '');
  const visibleLimit = 6;
  const initialCards = shouldLimitCards ? filtered.slice(0, visibleLimit) : filtered;
  const remainingCards = shouldLimitCards ? filtered.slice(visibleLimit) : [];
  
  const showAlphabetNav = activeCategories.size === 0 && searchQuery === '';
  alphabetNav.style.display = showAlphabetNav ? 'flex' : 'none';
  
  if (showAlphabetNav) {
    const groups = initialCards.reduce((acc, item) => { (acc[item.cat] = acc[item.cat] || []).push(item); return acc; }, {});

      Object.keys(remainingGroups).sort().forEach(cat => {

function initFilters() {
  const counts = {};
  data.forEach(i => counts[i.cat] = (counts[i.cat] || 0) + 1);
  const pinnedCats = ['Images', 'Video'];
  
  filterCollapsible.innerHTML = '';
  const oldBtns = filterBar.querySelectorAll('.filter-btn:not([data-cat="all"])');
  oldBtns.forEach(b => b.remove());

  pinnedCats.forEach(cat => {
     if (counts[cat]) {
        const btn = document.createElement('button');
        btn.className = 'filter-btn';
        btn.dataset.cat = cat;
        btn.innerHTML = `${cat}<span class="filter-remove-icon"><svg viewBox="0 0 24 24" fill="none" stroke-linecap="round" stroke-linejoin="round"><line x1="18" y1="6" x2="6" y2="18"></line><line x1="6" y1="6" x2="18" y2="18"></line></svg></span>`;
        filterBar.insertBefore(btn, filterCollapsible);
     }
  });

  const remainingCats = Object.keys(counts).filter(c => !pinnedCats.includes(c));
  remainingCats.sort((a, b) => counts[a] - counts[b]);

  remainingCats.forEach(cat => {
     const btn = document.createElement('button');
     btn.className = 'filter-btn';
     btn.dataset.cat = cat;
     btn.innerHTML = `${cat}<span class="filter-remove-icon"><svg viewBox="0 0 24 24" fill="none" stroke-linecap="round" stroke-linejoin="round"><line x1="18" y1="6" x2="6" y2="18"></line><line x1="6" y1="6" x2="18" y2="18"></line></svg></span>`;
     filterCollapsible.appendChild(btn);
  });

  if (remainingCats.length > 0) filterToggle.style.display = 'flex';
  else filterToggle.style.display = 'none';
  
  filterToggle.addEventListener('click', () => {
    filtersExpanded = !filtersExpanded;
    filterCollapsible.classList.toggle('show', filtersExpanded);
    filterToggle.classList.toggle('expanded', filtersExpanded);
    filterToggle.firstChild.textContent = filtersExpanded ? 'Show less' : 'Show more';
  });
  
  filterBar.addEventListener('click', (e) => {
    const btn = e.target.closest('.filter-btn');
    if (!btn) return;
    
    const cat = btn.dataset.cat;
    const isRemoveClick = e.target.closest('.filter-remove-icon');
    
    if (cat === 'all') {
       activeCategories.clear();
       updateFilterUI();
       showLoader();
       setTimeout(() => { render(); hideLoader(); }, 100);
       return;
    }
    
    if (activeCategories.has(cat) || isRemoveClick) activeCategories.delete(cat);
    else if (activeCategories.size < 5) activeCategories.add(cat);
    else return;
    
    updateFilterUI();
    showLoader();
    setTimeout(() => { render(); hideLoader(); }, 100);
  });
}

function updateFilterUI() {
  const allBtn = document.querySelector('.filter-btn[data-cat="all"]');
  if (allBtn) allBtn.classList.toggle('active', activeCategories.size === 0);
  
  filterBar.querySelectorAll('.filter-btn:not([data-cat="all"])').forEach(btn => {
    btn.classList.toggle('active', activeCategories.has(btn.dataset.cat));
  });
}

let searchTimeout;
searchInput.addEventListener('input', (e) => {
  clearTimeout(searchTimeout);
  showLoader();
  searchTimeout = setTimeout(() => { searchQuery = e.target.value.trim(); render(); hideLoader(); }, 200);
});

document.addEventListener('keydown', (e) => {
  if (e.key === '/' && document.activeElement !== searchInput) { e.preventDefault(); searchInput.focus(); }
  if (e.key === 'Escape' && document.activeElement === searchInput) { searchInput.value = ''; searchQuery = ''; searchInput.blur(); render(); }
});

const scrollSpy = () => {
  if (window.getComputedStyle(alphabetNav).display === 'none') return;
  const scrollPos = window.pageYOffset;
  const windowHeight = window.innerHeight;
  const docHeight = document.documentElement.scrollHeight;
  
  let currentCat = '';
  document.querySelectorAll('.category-section').forEach(s => {
    if (s.getBoundingClientRect().top < windowHeight / 2) currentCat = s.dataset.category;
  });

  if (currentCat && alphabetCursor) {
    const letter = currentCat.charAt(0).toUpperCase();
    if (alphabetCursor.querySelector('span').textContent !== letter) alphabetCursor.querySelector('span').textContent = letter;
  }

  const scrollPercent = scrollPos / (docHeight - windowHeight);
  const trackHeight = document.querySelector('.alphabet-track').offsetHeight;
  const cursorOffset = 14;
  let topPos = Math.max(cursorOffset, Math.min(scrollPercent * trackHeight, trackHeight - cursorOffset));
  alphabetCursor.style.top = `${topPos}px`;
};

window.addEventListener('scroll', scrollSpy, { passive: true });

function init() {
  showLoader();
  setTimeout(() => {
    initFilters();
    render();
    hideLoader();
    scrollSpy();
    initBannerRotation();
  }, 200);
}

if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
else init();

const group = new THREE.Group();
scene.add(group);

const icoGeo = new THREE.IcosahedronGeometry(12, 1);
const icoMesh = new THREE.Mesh(icoGeo, wireMat);
icoMesh.position.set(-10, 5, -20);
group.add(icoMesh);

const shapes = [];
for(let i=0; i<6; i++) {
    const geo = i % 2 === 0 ? new THREE.OctahedronGeometry(1.5, 0) : new THREE.TorusGeometry(1, 0.3, 16, 100);
    const mat = i % 2 === 0 ? yellowMat : glassMat;
    const mesh = new THREE.Mesh(geo, mat);
    mesh.position.set((Math.random() - 0.5) * 30, (Math.random() - 0.5) * 20, (Math.random() - 0.5) * 10);
    mesh.rotation.set(Math.random(), Math.random(), 0);
    mesh.userData = { rotSpeed: (Math.random() - 0.5) * 0.02, floatSpeed: 0.5 + Math.random() * 0.5 };
    shapes.push(mesh);
    scene.add(mesh);
}

scene.add(new THREE.AmbientLight(0xffffff, 0.5));
const spotLight = new THREE.SpotLight(0xffd966, 100);
spotLight.position.set(20, 30, 30);
spotLight.angle = 0.5;
scene.add(spotLight);

const pointLight = new THREE.PointLight(0xffffff, 20);
pointLight.position.set(-20, -20, 20);
scene.add(pointLight);

let time = 0;
function animate() {
    requestAnimationFrame(animate);
    time += 0.01;
    icoMesh.rotation.x += 0.001;
    icoMesh.rotation.y += 0.001;
    shapes.forEach(m => {
        m.rotation.x += m.userData.rotSpeed;
        m.rotation.y += m.userData.rotSpeed * 0.5;
        m.position.y += Math.sin(time * m.userData.floatSpeed) * 0.005;
    });
    renderer.render(scene, camera);
}
animate();

window.addEventListener('resize', () => {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
});
</script>
